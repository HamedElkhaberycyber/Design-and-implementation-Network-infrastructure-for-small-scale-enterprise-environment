----------------------------------------------------
####################################################
‚úÖ 1. Assign IP, enable SSH, and rename switches
####################################################
----------------------------------------------------
# Hostname and IP Setup (example for Floor 1 switch)
hostname Floor-1
interface vlan 1
  ip address 192.168.1.11 255.255.255.0
  no shutdown
exit

# Enable SSH and set credentials
ip ssh server
username admin privilege 15 password 0 strong_password_here
crypto key generate rsa
----------------------------------------------------
####################################################
‚úÖ 2. VLAN Configuration for core SW
####################################################
----------------------------------------------------
vlan 10
  name Floors
exit
vlan 20
  name QU_HR_IT
exit

# Assign VLANs to ports 
interface range gi11 - 15
  switchport mode access
  switchport access vlan 10
  spanning-tree portfast
exit

# For QU, HR, IT switch
interface range gi6 - 10
  switchport mode access
  switchport access vlan 20
  spanning-tree portfast
exit

----------------------------------------------------
####################################################
‚úÖ 3. Shutdown Ports 20-24
####################################################
----------------------------------------------------
interface range gi20 - 24
  shutdown
exit

----------------------------------------------------
####################################################
‚úÖ 4. Spanning Tree Protocol
####################################################
----------------------------------------------------
spanning-tree mode rapid-pvst
spanning-tree portfast default

----------------------------------------------------
####################################################
‚úÖ 5. Enable Port Security with Sticky MAC
####################################################
----------------------------------------------------
interface range gi1 - 20
  switchport port-security
  switchport port-security maximum 2
  switchport port-security mac-address sticky
  switchport port-security violation restrict
exit

----------------------------------------------------
####################################################
‚úÖ Trunk Port on Gi1/1 (Port 1)
####################################################
----------------------------------------------------
interface gi1/1
  switchport mode trunk
  switchport trunk allowed vlan 10,20
 
exit
----------------------------------------------------
####################################################
‚úÖ Enable SNMP for network monitoring tools
####################################################
----------------------------------------------------

snmp-server community public RO
snmp-server enable traps

----------------------------------------------------
####################################################
‚úÖ  NTP (Time Sync)
####################################################
----------------------------------------------------
ntp server 192.168.1.200  
----------------------------------------------------
####################################################
‚úÖ  DHCP Snooping
####################################################
----------------------------------------------------
ip dhcp snooping
ip dhcp snooping vlan 10,20,30
interface gi1/1
  ip dhcp snooping trust                    ! On trunk/uplink ports only
interface range gi2-19
  ip dhcp snooping limit rate 10
